X <- as.matrix(geno_raw)

common_genos <- intersect(rownames(X), pheno$Taxa)

# Compute genomic relationship matrix (G matrix)

A1 <- A.mat(
  X,

# Cross-validation setup
set.seed(123)

cycles <- 100
folds  <- 10

trait_names <- colnames(y)[-1]
n_traits <- length(trait_names)

cat("Cross-validation setup\n")
cat("  cycles:", cycles, "\n")
cat("  folds:", folds, "\n")
cat("  traits:", n_traits, "\n")

# Initialize storage for prediction accuracy
prediction_accuracy <- matrix(nrow = cycles * folds, ncol = n_traits)
colnames(prediction_accuracy) <- colnames(y)[-1]

all_gebvs <- list()

# Loop through cycles for cross-validation
for (cycle in 1:cycles) {
  if (cycle %% 10 == 0) cat("  Processing cycle", cycle, "of", cycles, "\n")
  
  # Create fold assignments (move this outside the fold loop)
  set.seed(cycle)  # Ensures different train/test splits across cycles
  fold_assignments <- sample(rep(1:folds, length.out = length(common_genos)))
  
  cycle_gebvs <- list()
  
  # Loop through traits for prediction
  for (trait_idx in 2:ncol(y)) {
    trait_name <- colnames(y)[trait_idx]
    
    # Loop through folds
    for (fold in 1:folds) {
      
      # Get test genotypes for this fold
      test_genos <- common_genos[fold_assignments == fold]
      
      # Create training dataset (mask test genotypes)
      y_train <- y
      y_train[y_train$Taxa %in% test_genos, trait_idx] <- NA
      
      # Run genomic prediction using rrBLUP
      gp_model <- kin.blup(data = y_train, 
                           geno = "Taxa", 
                           pheno = trait_name, 
                           K = A1)
      
      # Get predicted values for test set
      y_test_actual <- y[y$Taxa %in% test_genos, trait_idx]
      y_test_pred <- gp_model$g[test_genos]
      
      # Calculate prediction accuracy (correlation)
      accuracy <- cor(y_test_pred, y_test_actual, 
                      method = "pearson", 
                      use = "complete.obs")
      
      # Store accuracy
      row_idx <- (cycle - 1) * folds + fold
      prediction_accuracy[row_idx, trait_idx - 1] <- accuracy
      
      # Store GEBVs for the first cycle
      if (cycle == 1 && fold == 1) {
        cycle_gebvs[[trait_name]] <- gp_model$g
      }
    }
  }
  
  # Store GEBVs from the first cycle
  if (cycle == 1) {
    all_gebvs <- cycle_gebvs
  }
}

# Helper function to get the top k taxa for a given numeric vector
get_topk <- function(x, taxa, k = 10, decreasing = TRUE) {
  ord <- order(x, decreasing = decreasing, na.last = NA)
  taxa[ord][seq_len(min(k, sum(!is.na(x))))]  # Return top k genotypes
}

# Initialize an empty list to store results
res_list <- list()

# Loop over traits to calculate the Coincidence Index (CI)
for (tr in colnames(y)[-1]) {  # Loop through all traits (excluding the "Taxa" column)
  
  # Get the BLUP values and GEBV values for the trait
  blup_vals <- y[[tr]]
  gebv_vals <- all_gebvs[[tr]]
  
  # Get top k genotypes from BLUPs and GEBVs
  top_blup <- get_topk(blup_vals, taxa = y$Taxa, k = 10)  # top 10 based on BLUPs
  top_gebv <- get_topk(gebv_vals, taxa = y$Taxa, k = 10)  # top 10 based on GEBVs
  
  # Calculate the Coincidence Index (CI) - the percentage overlap between top BLUPs and top GEBVs
  ci_fun <- function(a, b) {
    # Calculate the intersection of the two sets and return the coincidence percentage
    if (any(is.na(a)) || any(is.na(b))) return(NA_real_)
    length(intersect(a, b)) / length(a) * 100
  }
  
  # Calculate CI for the current trait
  ci_blup_gebv <- ci_fun(top_blup, top_gebv)
  
  # Store results for this trait
  res_list[[tr]] <- data.frame(
    Trait = tr,
    CI_BLUP_GEBV = ci_blup_gebv,
    stringsAsFactors = FALSE
  )
}

    trait_name <- colnames(y)[trait_idx]

      test_genos <- common_genos[fold_assignments == fold]
      
      y_vec <- y[[trait_name]]
      y_train_vec <- y_vec
      y_train_vec[match(test_genos, common_genos)] <- NA
      
      fit <- mixed.solve(
        y = y_train_vec,
        K = A1,
        X = X_fixed_base
      )
      
      u_hat <- fit$u
      names(u_hat) <- common_genos
      
      y_fixed_hat <- as.vector(X_fixed_base %*% fit$beta)
      
      y_test_pred   <- y_fixed_hat[match(test_genos, common_genos)] + u_hat[test_genos]
      y_test_actual <- y_vec[match(test_genos, common_genos)]
      
      acc <- cor(y_test_pred, y_test_actual, use = "complete.obs")
      
      row_idx <- (cycle - 1) * folds + fold
      prediction_accuracy[row_idx, trait_idx - 1] <- acc
      
      if (cycle == 1 && fold == 1) {
        trait_gebv <- y_fixed_hat + u_hat
        names(trait_gebv) <- common_genos
        cycle_gebvs[[trait_name]] <- trait_gebv
      }
    }
  }
  
  if (cycle == 1) {
    all_gebvs <- cycle_gebvs
  }
}

  top_gebv <- get_topk(gebv_vals, gebv_wg$Taxa, k = 10)
  
  ci_val <- length(intersect(top_blup, top_gebv)) / length(top_blup) * 100
  
  res_ci_wg[[tr]] <- data.frame(
    Trait = tr,
    CI_BLUP_GEBV = ci_val,
    stringsAsFactors = FALSE
  )
}
ci_wg <- bind_rows(res_ci_wg)
ci_wg$Model <- "GBLUP"
write.csv(ci_wg, "WGP_CI_BLUP_GEBV_results.csv", row.names = FALSE)
print(ci_wg)
